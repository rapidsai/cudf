57 thrust::device_vector<index_type>
58 create_missing_indices(
59  index_type const * const index_ptr,
60  const size_type max_index_value,
61  const size_type index_size) {
62  //Assume all the indices in invalid_index_map are invalid
63  thrust::device_vector<index_type> invalid_index_map(max_index_value, 1);
64  //Vector allocated for unmatched result
65  thrust::device_vector<index_type> unmatched_indices(max_index_value);
66  //Functor to check for index validity since left joins can create invalid indices
67  ValidRange<size_type> valid_range(0, max_index_value);
68 
69  //invalid_index_map[index_ptr[i]] = 0 for i = 0 to max_index_value
70  //Thus specifying that those locations are valid
71  thrust::scatter_if(
72  thrust::device,
73  thrust::make_constant_iterator(0),
74  thrust::make_constant_iterator(0) + index_size,
75  index_ptr,//Index locations
76  index_ptr,//Stencil - Check if index location is valid
77  invalid_index_map.begin(),//Output indices
78  valid_range);//Stencil Predicate
79  size_type begin_counter = static_cast<size_type>(0);
80  size_type end_counter = static_cast<size_type>(invalid_index_map.size());
81  //Create list of indices that have been marked as invalid
82  size_type compacted_size = thrust::copy_if(
83  thrust::device,
84  thrust::make_counting_iterator(begin_counter),
85  thrust::make_counting_iterator(end_counter),
86  invalid_index_map.begin(),
87  unmatched_indices.begin(),
88  thrust::identity<index_type>()) -
89  unmatched_indices.begin();
90  unmatched_indices.resize(compacted_size);
91  return unmatched_indices;
92 }

109 template <typename data_type, typename size_type>
110 gdf_error expand_buffer(
111  data_type ** buffer,
112  size_type * const buffer_capacity,
113  const size_type buffer_size,
114  const size_type expand_size) {
115  size_type requested_size = buffer_size + expand_size;
116  //No need to proceed if the buffer can contain requested additional elements
117  if (*buffer_capacity >= requested_size) {
118  return GDF_SUCCESS;
119  }
120  data_type * new_buffer{nullptr};
121  data_type * old_buffer = *buffer;
122  RMM_TRY( RMM_ALLOC((void**)&new_buffer, requested_size*sizeof(data_type), 0) );
123  CUDA_TRY( cudaMemcpy(new_buffer, old_buffer, buffer_size*sizeof(data_type), cudaMemcpyDeviceToDevice) );
124  RMM_TRY( RMM_FREE(old_buffer, 0) );
125  *buffer = new_buffer;
126  *buffer_capacity = requested_size;
127 
128  return GDF_SUCCESS;
129 }

148 template <typename index_type, typename size_type>
149 gdf_error append_full_join_indices(
150  index_type ** l_index_ptr,
151  index_type ** r_index_ptr,
152  size_type * const index_capacity,
153  size_type * const index_size,
154  const size_type max_index_value) {
155  gdf_error err;
156  //Get array of indices that do not appear in r_index_ptr
157  thrust::device_vector<index_type> unmatched_indices =
158  create_missing_indices(
159  *r_index_ptr, max_index_value, *index_size);
160  CUDA_CHECK_LAST()
161 
162  //Expand l_index_ptr and r_index_ptr if necessary
163  size_type mismatch_index_size = unmatched_indices.size();
164  size_type l_index_capacity = *index_capacity;
165  size_type r_index_capacity = *index_capacity;
166  err = expand_buffer(l_index_ptr, &l_index_capacity, *index_size, mismatch_index_size);
167  if (GDF_SUCCESS != err) return err;
168  err = expand_buffer(r_index_ptr, &r_index_capacity, *index_size, mismatch_index_size);
169  if (GDF_SUCCESS != err) return err;
170 
171  //Copy JoinNoneValue to l_index_ptr to denote that a match does not exist on the left
172  thrust::fill(
173  thrust::device,
174  *l_index_ptr + *index_size,
175  *l_index_ptr + *index_size + mismatch_index_size,
176  JoinNoneValue);
177 
178  //Copy unmatched indices to the r_index_ptr
179  thrust::copy(thrust::device,
180  unmatched_indices.begin(),
181  unmatched_indices.begin() + mismatch_index_size,
182  *r_index_ptr + *index_size);
183  *index_capacity = l_index_capacity;
184  *index_size = *index_size + mismatch_index_size;
185 
186  CUDA_CHECK_LAST()
187  return GDF_SUCCESS;
188 }

206 template <JoinType join_type,
207  typename multimap_type,
208  typename size_type>
209 gdf_error estimate_join_output_size(gdf_table<size_type> const & build_table,
210  gdf_table<size_type> const & probe_table,
211  multimap_type const & hash_table,
212  size_type * join_output_size_estimate)
213 {
214  const size_type build_table_num_rows{build_table.get_column_length()};
215  const size_type probe_table_num_rows{probe_table.get_column_length()};
216 
217  // If the probe table is significantly larger (5x) than the build table,
218  // then we attempt to only use a subset of the probe table rows to compute an
219  // estimate of the join output size.
220  size_type probe_to_build_ratio{0};
221  if(build_table_num_rows > 0) {
222  probe_to_build_ratio = static_cast<size_type>(std::ceil(static_cast<float>(probe_table_num_rows)/build_table_num_rows));
223  }
224  else {
225  // If the build table is empty, we know exactly how large the output
226  // will be for the different types of joins and can return immediately
227  switch(join_type)
228  {
229  case JoinType::INNER_JOIN:
230  {
231  // Inner join with an empty table will have no output
232  *join_output_size_estimate = 0;
233  break;
234  }
235  case JoinType::LEFT_JOIN:
236  {
237  // Left join with an empty table will have an output of NULL rows
238  // equal to the number of rows in the probe table
239  *join_output_size_estimate = probe_table_num_rows;
240  break;
241  }
242  default:
243  return GDF_UNSUPPORTED_JOIN_TYPE;
244  }
245  return GDF_SUCCESS;
246  }
247 
248  size_type sample_probe_num_rows{probe_table_num_rows};
249  constexpr size_type MAX_RATIO{5};
250  if(probe_to_build_ratio > MAX_RATIO)
251  {
252  sample_probe_num_rows = build_table_num_rows;
253  }
254 
255  // Allocate storage for the counter used to get the size of the join output
256  size_type * d_size_estimate{nullptr};
257  size_type h_size_estimate{0};
258 
259  CUDA_TRY(cudaMallocHost(&d_size_estimate, sizeof(size_type)));
260  *d_size_estimate = 0;
261 
262  CUDA_TRY( cudaGetLastError() );
263 
264  // Continue probing with a subset of the probe table until either:
265  // a non-zero output size estimate is found OR
266  // all of the rows in the probe table have been sampled
267  do{
268 
269  sample_probe_num_rows = std::min(sample_probe_num_rows, probe_table_num_rows);
270 
271  *d_size_estimate = 0;
272 
273  constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
274  const size_type probe_grid_size{(sample_probe_num_rows + block_size -1)/block_size};
275 
276  // Probe the hash table without actually building the output to simply
277  // find what the size of the output will be.
278  compute_join_output_size<join_type,
279  multimap_type,
280  size_type,
281  block_size,
282  DEFAULT_CUDA_CACHE_SIZE>
283  <<<probe_grid_size, block_size>>>(&hash_table,
284  build_table,
285  probe_table,
286  sample_probe_num_rows,
287  d_size_estimate);
288 
289  // Device sync is required to ensure d_size_estimate is updated
290  CUDA_TRY( cudaDeviceSynchronize() );
291 
292 
293  // Increase the estimated output size by a factor of the ratio between the
294  // probe and build tables
295  h_size_estimate = *d_size_estimate * probe_to_build_ratio;
296 
297  // If the size estimate is non-zero, then we have a valid estimate and can break
298  // If sample_probe_num_rows >= probe_table_num_rows, then we've sampled the entire
299  // probe table, in which case the estimate is exact and we can break
300  if((h_size_estimate > 0)
301  || (sample_probe_num_rows >= probe_table_num_rows))
302  {
303  break;
304  }
305 
306  // If the size estimate is zero, then double the number of sampled rows in the probe
307  // table. Reduce the ratio of the number of probe rows sampled to the
308  // number of rows in the build table by the same factor
309  if(0 == h_size_estimate)
310  {
311  constexpr size_type GROW_RATIO{2};
312  sample_probe_num_rows *= GROW_RATIO;
313  probe_to_build_ratio = static_cast<size_type>(std::ceil(static_cast<float>(probe_to_build_ratio)/GROW_RATIO));
314  }
315 
316  } while(true);
317 
318  CUDA_TRY( cudaFreeHost(d_size_estimate) );
319 
320  *join_output_size_estimate = h_size_estimate;
321 
322  return GDF_SUCCESS;
323 }

371 gdf_error compute_hash_join(
372  gdf_column * const output_l,
373  gdf_column * const output_r,

382 
383  // The LEGACY allocator allocates the hash table array with normal cudaMalloc,
384  // the non-legacy allocator uses managed memory
385 #ifdef HT_LEGACY_ALLOCATOR
386  using multimap_type = concurrent_unordered_multimap<hash_value_type,
387  output_index_type,
388  size_type,
389  std::numeric_limits<hash_value_type>::max(),
390  std::numeric_limits<output_index_type>::max(),
391  default_hash<hash_value_type>,
392  equal_to<hash_value_type>,
393  legacy_allocator< thrust::pair<hash_value_type, output_index_type> > >;
394 #else
395  using multimap_type = concurrent_unordered_multimap<hash_value_type,
396  output_index_type,
397  size_type,
398  std::numeric_limits<hash_value_type>::max(),
399  std::numeric_limits<output_index_type>::max()>;
400 #endif
401  //If FULL_JOIN is selected then we process as LEFT_JOIN till we need to take care of unmatched indices
402  constexpr JoinType base_join_type = (join_type == JoinType::FULL_JOIN)? JoinType::LEFT_JOIN : join_type;
403 
404  // Hash table will be built on the right table
405  gdf_table<size_type> const & build_table{right_table};
406  const size_type build_table_num_rows{build_table.get_column_length()};
407 
408  // Probe with the left table
409  gdf_table<size_type> const & probe_table{left_table};
410  const size_type probe_table_num_rows{probe_table.get_column_length()};
411 
412  // Hash table size must be at least 1 in order to have a valid allocation.
413  // Even if the hash table will be empty, it still must be allocated for the
414  // probing phase in the event of an outer join
415  size_t const hash_table_size =
416  std::max(compute_hash_table_size(build_table_num_rows), size_t{1});
417 
418  std::unique_ptr<multimap_type> hash_table(new multimap_type(hash_table_size));
419 
420  // FIXME: use GPU device id from the context?
421  // but moderngpu only provides cudaDeviceProp
422  // (although should be possible once we move to Arrow)
423  hash_table->prefetch(0);
424 
425  CUDA_TRY( cudaDeviceSynchronize() );
426 
427  // Allocate a gdf_error for the device to hold error code returned from
428  // the build kernel and intialize with GDF_SUCCESS
429  // Use Page Locked memory to avoid overhead of memcpys
430  gdf_error * d_gdf_error_code{nullptr};
431  CUDA_TRY( cudaMallocHost(&d_gdf_error_code, sizeof(gdf_error)) );
432  *d_gdf_error_code = GDF_SUCCESS;
433 
434  constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
435 
436  // build the hash table
437  if(build_table_num_rows > 0)
438  {
439  const size_type build_grid_size{(build_table_num_rows + block_size - 1)/block_size};
440  build_hash_table<<<build_grid_size, block_size>>>(hash_table.get(),
441  build_table,
442  build_table_num_rows,
443  d_gdf_error_code);
444 
445  // Device synch is required to ensure d_gdf_error_code
446  // has been written
447  CUDA_TRY( cudaDeviceSynchronize() );
448  }
449 
450  // Check error code from the kernel
451  gdf_error_code = *d_gdf_error_code;
452  if(GDF_SUCCESS != gdf_error_code){
453  return gdf_error_code;
454  }
455 
456 
457  size_type estimated_join_output_size{0};
458  gdf_error_code = estimate_join_output_size<base_join_type, multimap_type>(build_table, probe_table, *hash_table, &estimated_join_output_size);
459 
460  if(GDF_SUCCESS != gdf_error_code){
461  return gdf_error_code;
462  }
463 
464  // If the estimated output size is zero, return immediately
465  if(0 == estimated_join_output_size){
466  return GDF_SUCCESS;
467  }
468 
469  // Because we are approximating the number of joined elements, our approximation
470  // might be incorrect and we might have underestimated the number of joined elements.
471  // As such we will need to de-allocate memory and re-allocate memory to ensure
472  // that the final output is correct.
473  size_type h_actual_found{0};
474  output_index_type *output_l_ptr{nullptr};
475  output_index_type *output_r_ptr{nullptr};
476  bool cont = true;
477 
478  // Allocate device global counter used by threads to determine output write location
479  size_type *d_global_write_index{nullptr};
480  RMM_TRY( RMM_ALLOC((void**)&d_global_write_index, sizeof(size_type), 0) ); // TODO non-default stream?
481 
482  // Because we only have an estimate of the output size, we may need to probe the
483  // hash table multiple times until we've found an output buffer size that is large enough
484  // to hold the output
485  while(cont)
486  {
487  output_l_ptr = nullptr;
488  output_r_ptr = nullptr;
489 
490  // Allocate temporary device buffer for join output
491  RMM_TRY( RMM_ALLOC((void**)&output_l_ptr, estimated_join_output_size*sizeof(output_index_type), 0) );
492  RMM_TRY( RMM_ALLOC((void**)&output_r_ptr, estimated_join_output_size*sizeof(output_index_type), 0) );
493  CUDA_TRY( cudaMemsetAsync(d_global_write_index, 0, sizeof(size_type), 0) );
494 
495  const size_type probe_grid_size{(probe_table_num_rows + block_size -1)/block_size};
496 
497  // Do the probe of the hash table with the probe table and generate the output for the join
498  probe_hash_table<base_join_type,
499  multimap_type,
500  hash_value_type,
501  size_type,
502  output_index_type,
503  block_size,
504  DEFAULT_CUDA_CACHE_SIZE>
505  <<<probe_grid_size, block_size>>> (hash_table.get(),
506  build_table,
507  probe_table,
508  probe_table.get_column_length(),
509  output_l_ptr,
510  output_r_ptr,
511  d_global_write_index,
512  estimated_join_output_size,
513  flip_results);
514 
515  CUDA_TRY( cudaGetLastError() );
516 
517  CUDA_TRY( cudaMemcpy(&h_actual_found, d_global_write_index, sizeof(size_type), cudaMemcpyDeviceToHost));
518 
519  // The estimate was too small. Double the estimate and try again
520  if(estimated_join_output_size < h_actual_found){
521  cont = true;
522  estimated_join_output_size *= 2;
523  // Free the old buffers to prevent a memory leak on the new allocation
524  RMM_TRY( RMM_FREE(output_l_ptr, 0) );
525  RMM_TRY( RMM_FREE(output_r_ptr, 0) );
526  }
527  else
528  {
529  cont = false;
530  }
531  }
532 
533  // free memory used for the counters
534  RMM_TRY( RMM_FREE(d_global_write_index, 0) );
535 
536  if (join_type == JoinType::FULL_JOIN) {
537  append_full_join_indices(
538  &output_l_ptr, &output_r_ptr,
539  &estimated_join_output_size,
540  &h_actual_found, build_table_num_rows);

542 
543  // If the estimated join output size was larger than the actual output size,
544  // then the buffers are larger than necessary. Allocate buffers of the actual
545  // output size and copy the results to the buffers of the correct size
546  // FIXME Is this really necessary? It's probably okay to have the buffers be oversized
547  // and avoid the extra allocation/memcopy
548  if (estimated_join_output_size > h_actual_found) {

550  output_index_type *copy_output_r_ptr{nullptr};

555  RMM_TRY( RMM_FREE(output_l_ptr, 0) );

558  output_r_ptr = copy_output_r_ptr;
559  }
560 
561  // Free the device error code
562  CUDA_TRY( cudaFreeHost(d_gdf_error_code) );
563 
564  // Deduce the type of the output gdf_columns
565  gdf_dtype dtype;
566  switch(sizeof(output_index_type))
567  {
568  case 1 : dtype = GDF_INT8; break;
569  case 2 : dtype = GDF_INT16; break;
570  case 4 : dtype = GDF_INT32; break;
571  case 8 : dtype = GDF_INT64; break;
572  }
573  gdf_column_view(output_l, output_l_ptr, nullptr, h_actual_found, dtype);
574  gdf_column_view(output_r, output_r_ptr, nullptr, h_actual_found, dtype);
575 
576  return gdf_error_code;
577 }
