141 gdf_error GroupbyHash(gdf_table<size_type> const & groupby_input_table,
142  const aggregation_type * const in_aggregation_column,
143  gdf_table<size_type> & groupby_output_table,
144  aggregation_type * out_aggregation_column,
145  size_type * out_size,
146  aggregation_operation aggregation_op,
147  bool sort_result = false)
148 {
149  const size_type input_num_rows = groupby_input_table.get_column_length();
150 
151  // The map will store (row index, aggregation value)
152  // Where row index is the row number of the first row to be successfully inserted
153  // for a given unique row
154  using map_type = concurrent_unordered_map<size_type,
155  aggregation_type,
156  std::numeric_limits<size_type>::max(),
157  default_hash<size_type>,
158  equal_to<size_type>,
159  legacy_allocator<thrust::pair<size_type, aggregation_type> > >;
160 
161  // The hash table occupancy and the input size determines the size of the hash table
162  // e.g., for a 50% occupancy, the size of the hash table is twice that of the input
163  const size_type hash_table_size = static_cast<size_type>((static_cast<uint64_t>(input_num_rows) * 100 / DEFAULT_HASH_TABLE_OCCUPANCY));
164 
165  // Initialize the hash table with the aggregation operation functor's identity value
166  std::unique_ptr<map_type> the_map(new map_type(hash_table_size, aggregation_operation::IDENTITY));
167 
168  const dim3 build_grid_size ((input_num_rows + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
169  const dim3 block_size (THREAD_BLOCK_SIZE, 1, 1);
170 
171  CUDA_TRY(cudaGetLastError());
172 
173  // Inserts (i, aggregation_column[i]) as a key-value pair into the
174  // hash table. When a given key already exists in the table, the aggregation operation
175  // is computed between the new and existing value, and the result is stored back.
176  build_aggregation_table<<<build_grid_size, block_size>>>(the_map.get(),
177  groupby_input_table,
178  in_aggregation_column,
179  input_num_rows,
180  aggregation_op,
181  row_comparator<map_type, size_type>(*the_map, groupby_input_table, groupby_input_table));
182  CUDA_TRY(cudaGetLastError());
183 
184  // Used by threads to coordinate where to write their results
185  size_type * global_write_index{nullptr};
186  RMM_TRY(RMM_ALLOC((void**)&global_write_index, sizeof(size_type), 0)); // TODO: non-default stream?
187  CUDA_TRY(cudaMemset(global_write_index, 0, sizeof(size_type)));
188 
189  const dim3 extract_grid_size ((hash_table_size + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
190 
191  // Extracts every non-empty key and value into separate contiguous arrays,
192  // which provides the result of the groupby operation
193  extract_groupby_result<<<extract_grid_size, block_size>>>(the_map.get(),
194  hash_table_size,
195  groupby_output_table,
196  groupby_input_table,
197  out_aggregation_column,
198  global_write_index);
199 
200  CUDA_TRY(cudaGetLastError());
201 
202  // At the end of the extraction kernel, the global write index will be equal to
203  // the size of the output. Update the output size.
204  CUDA_TRY( cudaMemcpy(out_size, global_write_index, sizeof(size_type), cudaMemcpyDeviceToHost) );
205  RMM_TRY( RMM_FREE(global_write_index, 0) );
206  groupby_output_table.set_column_length(*out_size);
207 
208  // Optionally sort the groupby/aggregation result columns
209  if(true == sort_result) {
210 
211  rmm::device_vector<int32_t> sorted_indices(*out_size);
212  thrust::sequence(rmm::exec_policy()->on(0), sorted_indices.begin(), sorted_indices.end());
213 
214  gdf_column sorted_indices_col;
215  gdf_error status = gdf_column_view(&sorted_indices_col, (void*)thrust::raw_pointer_cast(sorted_indices.data()),
216  nullptr, *out_size, GDF_INT32);
217  if (status != GDF_SUCCESS)
218  return status;
219 
220  status = gdf_order_by(groupby_output_table.get_columns(), //input columns
221  nullptr,
222  groupby_output_table.get_num_columns(), //number of columns in the first parameter (e.g. number of columsn to sort by)
223  &sorted_indices_col, //a gdf_column that is pre allocated for storing sorted indices
224  0); //flag to indicate if nulls are to be considered smaller than non-nulls or viceversa
225  if (status != GDF_SUCCESS)
226  return status;
227 
228  groupby_output_table.gather(sorted_indices);
229 
230  rmm::device_vector<aggregation_type> agg(*out_size);
231  thrust::gather(rmm::exec_policy()->on(0),
232  sorted_indices.begin(), sorted_indices.end(),
233  out_aggregation_column,
234  agg.begin());
235  thrust::copy(rmm::exec_policy()->on(0), agg.begin(), agg.end(), out_aggregation_column);
236  }
237 
238  return GDF_SUCCESS;
239 }
