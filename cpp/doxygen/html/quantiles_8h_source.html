33 template<typename T>
34 size_t quantile_index(T* dv, size_t n, double q, double& fract_pos, cudaStream_t stream, bool flag_sorted)
35 {
36  if( !flag_sorted )
37  thrust::sort(rmm::exec_policy(stream)->on(stream), dv, dv+n);
38 
39  double pos = q*static_cast<double>(n);//(n-1);
40  size_t k = static_cast<size_t>(pos);
41 
42  fract_pos = pos - static_cast<double>(k);
43  if( k > 0 )//using n and (k-1) gives more intuitive result than using (n-1) and k
44  --k;
45 
46  return k;
47 }

101 template<typename VType,
102  typename RetT = double> // just in case double won't be enough to hold result, in the future
103 gdf_error select_quantile(VType* dv,
104  size_t n,
105  double q,
106  gdf_quantile_method prec,
107  RetT& result,
108  bool flag_sorted = false,
109  cudaStream_t stream = NULL)
110 {
111  using FctrType = std::function<RetT(VType, VType, double)>;
112  FctrType lin_interp{[](VType y0, VType y1, double x){
113  return static_cast<RetT>(static_cast<double>(y0) + x*static_cast<double>(y1-y0));//(f(x) - y0) / (x - 0) = m = (y1 - y0)/(1 - 0)
114  }};
115 
116  FctrType midpoint{[](VType y0, VType y1, double x){
117  return static_cast<RetT>(static_cast<double>(y0 + y1)/2.0);
118  }};
119 
120  FctrType nearest{[](VType y0, VType y1, double x){
121  return static_cast<RetT>(x < 0.5 ? y0 : y1);
122  }};
123 
124  FctrType lowest{[](VType y0, VType y1, double x){
125  return static_cast<RetT>(y0);
126  }};
127 
128  FctrType highest{[](VType y0, VType y1, double x){
129  return static_cast<RetT>(y1);
130  }};
131  FctrType fctr;
132  switch( prec )
133  {
134  case GDF_QUANT_LINEAR:
135  fctr = lin_interp;
136  break;
137 
138  case GDF_QUANT_LOWER:
139  fctr = lowest;
140  break;
141 
142  case GDF_QUANT_HIGHER:
143  fctr = highest;
144  break;
145 
146  case GDF_QUANT_MIDPOINT:
147  fctr = midpoint;
148  break;
149 
150  case GDF_QUANT_NEAREST:
151  fctr = nearest;
152  break;
153 
154  default:
155  return GDF_UNSUPPORTED_METHOD;
156  }
157  result = quantile_exact(dv, n, q, fctr, stream, flag_sorted);
158  return GDF_SUCCESS;
159 }
